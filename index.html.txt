localforage.config({name:'jingxinPK'})
let data={my:{name:'甲',records:[]},friend:{name:'乙',records:[]},pkDuration:30,startDate:null,reminders:[],currentUser:'my',roomId:null,fbUrl:null}
let timer={duration:600,remaining:600,interval:null,isRunning:false}
let roomRef=null

async function saveLocal(){await localforage.setItem('jingxinPK',data)}
async function loadLocal(){
  const stored=await localforage.getItem('jingxinPK')
  if(stored){data=stored;renderRecords();renderReminders()}
}
loadLocal()

// ---------- 开始PK ----------
function startPK(){
  data.my.name=document.getElementById('my-name').value||'甲'
  data.friend.name=document.getElementById('friend-name').value||'乙'
  data.pkDuration=parseInt(document.getElementById('pk-duration').value)
  const dateVal=document.getElementById('start-date').value
  data.startDate=dateVal?new Date(dateVal).toISOString():new Date().toISOString()
  data.roomId=document.getElementById('room-id').value
  saveLocal()
  alert('PK 已开始')
}

// ---------- 计时器 ----------
function toggleTimer(){
  const audio=document.getElementById('bowl-audio')
  if(timer.isRunning){
    clearInterval(timer.interval)
    timer.isRunning=false
    document.getElementById('timer-btn').textContent='开始'
    audio.play()
  }else{
    timer.isRunning=true
    document.getElementById('timer-btn').textContent='暂停'
    audio.play()
    timer.interval=setInterval(()=>{
      timer.remaining--
      if(timer.remaining<=0){clearInterval(timer.interval);timer.isRunning=false;audio.play()}
      renderTimer()
    },1000)
  }
}
function resetTimer(){clearInterval(timer.interval);timer.remaining=timer.duration;timer.isRunning=false;renderTimer()}
function renderTimer(){const m=Math.floor(timer.remaining/60);const s=timer.remaining%60;document.getElementById('timer-display').textContent=`${String(m).padStart(2,'0')}:${String(s).padStart(2,'0')}`}
renderTimer()

// ---------- 快速打卡 ----------
function quickLog(min){
  const user=data.currentUser==='my'?data.my:data.friend
  const rec={id:crypto.randomUUID(),minutes:min,date:new Date().toISOString()}
  user.records.push(rec)
  saveLocal()
  pushToCloud()
  renderRecords()
  checkReminders(user)
}

// ---------- 提醒 ----------
function sendReminder(){
  const reminder={id:crypto.randomUUID(),from:data.currentUser==='my'?data.my.name:data.friend.name,to:data.currentUser==='my'?data.friend.name:data.my.name,time:new Date().toISOString(),done:false}
  data.reminders.push(reminder)
  saveLocal()
  pushToCloud()
  renderReminders()
}
function renderReminders(){
  const ul=document.getElementById('reminderList')
  ul.innerHTML=''
  data.reminders.slice(-20).forEach(r=>{
    const li=document.createElement('li')
    li.className='list-item '+(r.done?'done':'')
    li.textContent=`${r.from} → ${r.to} ${new Date(r.time).toLocaleTimeString()} ${r.done?'✅已打卡':'⬜未打卡'}`
    ul.appendChild(li)
  })
}
function checkReminders(user){
  data.reminders.forEach(r=>{if(!r.done && r.to===user.name) r.done=true})
  renderReminders()
}

// ---------- 打卡记录 ----------
function renderRecords(){
  const ul=document.getElementById('recordList')
  ul.innerHTML=''
  const all=[...data.my.records,...data.friend.records].sort((a,b)=>new Date(b.date)-new Date(a.date)).slice(0,20)
  all.forEach(r=>{
    const li=document.createElement('li')
    li.className='list-item'
    li.textContent=`${r.date.split('T')[0]} ${r.minutes} 分钟`
    ul.appendChild(li)
  })
}
